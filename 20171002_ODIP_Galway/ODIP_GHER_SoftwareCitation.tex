\begin{frame}
\frametitle{Software tools and methods: Research platforms}

\onslide*<1>{	

\begin{description}
\item[Definition:] online infrastructures whose objective is to persistently store and archive digital artifacts\\
 relevant to research:
\begin{itemize}
\footnotesize
\item articles
\item data
\item images
\item code
\item \ldots
\end{itemize}
\end{description}

}

\onslide*<2>{

\href{https://figshare.com}{Figshare}: "improve the organization of users' research"
\begin{itemize}
\footnotesize

\item file upload, collaborative spaces, DOI attribution, \ldots 
\item on the platform: figures, datasets, media (including video), papers (including pre-prints), posters, code, and filesets.
\end{itemize}

\begin{figure}
\centering
\includegraphics[width=.9\textwidth]{figshare_example.png}\\
\scriptsize \url{https://figshare.com/articles/CNR-ISMAR_in_situ_observations_network/4001448}
\end{figure}

}

\onslide*<3>{

\href{http://www.dspace.org/}{DSpace}
\begin{itemize}
\footnotesize
\item has to be installed on a server manager by the search institution
\item Sandbox: \url{http://demo.dspace.org/}
\end{itemize}
\vspace{.5cm}

According to the \href{http://roar.eprints.org/}{Registry of Open Access Repositories (ROAR)}
\begin{description}
\item[1759] institutions or companies as users
\item[1374] repositories dedicated to Institutional or Departmental Research
\item[32] repositories dedicated to Research Data
\item[3] dedicated to Open and Linked Data
\end{description}
}

\onslide*<4>{
\href{https://ckan.org/}{Comprehensive Knowledge Archive Network (CKAN)}:\\
can be considered as a tool for making open data websites
\begin{itemize}
\footnotesize
\item streamline publishing, sharing, finding and using data
\item used by numerous governments, organisations and communities around the world
\item has to be installed on a server manager by the search institution
\end{itemize}
}

\onslide*<5>{
\href{https://zenodo.org/}{Zenodo}: research data repository\\
funded by the European Commission
\begin{itemize}
\footnotesize
\item ingest all research outputs and any file format
\item DOIs assigned to have uniquely citable files
\item integrated into reporting lines for research via OpenAIRE.
\end{itemize}

\begin{figure}
\centering
\includegraphics[width=.8\textwidth]{zenodo_matplotlib}
\end{figure}

}
